import React, { useState, useEffect, useCallback } from 'react'; import { initializeApp } from 'firebase/app'; import { getAuth, GoogleAuthProvider, signInWithPopup, onAuthStateChanged, signOut, signInWithCustomToken, signInAnonymously } from 'firebase/auth'; import { getFirestore, doc, getDoc, setDoc, collection, query, where, getDocs } from 'firebase/firestore'; // --- Global Context from Canvas Environment (MANDATORY) --- const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : '{}'); const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id'; const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null; // --- Firebase Paths --- const USER_PROFILES_COLLECTION = `artifacts/${appId}/public/data/user_profiles`; const USER_ID_MAP_COLLECTION = `artifacts/${appId}/public/data/user_id_map`; // Styles for the UI (using Tailwind CSS classes) const containerClasses = "min-h-screen bg-gray-900 text-white flex items-center justify-center p-4 font-inter"; const cardClasses = "bg-gray-800 p-8 rounded-xl shadow-2xl w-full max-w-md transition-all duration-300"; const buttonClasses = "w-full py-3 px-4 rounded-lg font-bold transition-all duration-200 shadow-md"; /** * Main application component. * Manages the authentication state and navigation flow. */ const App = () => { const [db, setDb] = useState(null); const [auth, setAuth] = useState(null); const [user, setUser] = useState(null); // Firebase user object const [profile, setProfile] = useState(null); // Firestore profile data const [appState, setAppState] = useState('LOADING'); // 'LOADING', 'SIGN_IN', 'CREATE_ID', 'DASHBOARD' const [loadingMessage, setLoadingMessage] = useState('Starting app...'); const [usernameInput, setUsernameInput] = useState(''); const [usernameError, setUsernameError] = useState(''); // 1. Initialize Firebase and Auth State Listener useEffect(() => { try { if (Object.keys(firebaseConfig).length === 0) { throw new Error("Firebase config is missing or empty."); } const firebaseApp = initializeApp(firebaseConfig); const firestore = getFirestore(firebaseApp); const firebaseAuth = getAuth(firebaseApp); setDb(firestore); setAuth(firebaseAuth); // Authenticate with custom token if available, otherwise sign in anonymously const signInUser = async () => { setLoadingMessage('Authenticating...'); try { if (initialAuthToken) { await signInWithCustomToken(firebaseAuth, initialAuthToken); } else { await signInAnonymously(firebaseAuth); } } catch (error) { console.error("Authentication failed:", error); // Fallback to anonymous if custom token fails await signInAnonymously(firebaseAuth); } }; signInUser(); // Listen for auth state changes const unsubscribe = onAuthStateChanged(firebaseAuth, (currentUser) => { setUser(currentUser); if (currentUser) { // A user is signed in (either Google or Anonymous) setLoadingMessage('Checking profile...'); checkProfile(firestore, currentUser.uid); } else { // No user (initial state or sign out) setAppState('SIGN_IN'); } }); return () => unsubscribe(); } catch (e) { console.error("Firebase initialization error:", e); setLoadingMessage('Error: Cannot initialize Firebase.'); setAppState('SIGN_IN'); // Show sign-in as a fallback } }, []); // 2. Check Firestore Profile for @user_id const checkProfile = useCallback(async (firestore, uid) => { if (!uid) return; try { const profileRef = doc(firestore, USER_PROFILES_COLLECTION, uid); const profileSnap = await getDoc(profileRef); if (profileSnap.exists() && profileSnap.data().uniqueId) { // SUCCESS: User has an @user_id, proceed to dashboard setProfile(profileSnap.data()); setAppState('DASHBOARD'); setLoadingMessage('Welcome back!'); } else { // NEW USER FLOW: User signed in but needs to create an @user_id setAppState('CREATE_ID'); } } catch (error) { console.error("Error checking profile:", error); setAppState('CREATE_ID'); // Default to ID creation on error } }, []); // 3. Google Sign-In Handler const handleGoogleSignIn = async () => { if (!auth) return; setLoadingMessage('Signing in with Google...'); setAppState('LOADING'); try { const provider = new GoogleAuthProvider(); // Ensure Google sign-in is possible (not anonymous) if (user && user.isAnonymous) { await signOut(auth); } await signInWithPopup(auth, provider); // onAuthStateChanged will handle the state transition } catch (error) { console.error("Google Sign-In failed:", error); // Handle 'auth/popup-closed-by-user' or other errors setAppState('SIGN_IN'); alertUser('Sign-in Failed', error.message); } }; // 4. Check for Unique @user_id Availability const isUsernameUnique = async (username) => { if (!db) return false; // Sanitize and validate username const sanitizedUsername = username.toLowerCase().trim(); if (sanitizedUsername.length < 3 || !/^[a-z0-9_]+$/.test(sanitizedUsername)) { setUsernameError('ID must be 3+ characters and contain only lowercase letters, numbers, or underscores.'); return false; } setLoadingMessage('Checking ID uniqueness...'); // Query the user_id_map collection const idMapRef = collection(db, USER_ID_MAP_COLLECTION); const q = query(idMapRef, where('uniqueId', '==', sanitizedUsername)); try { // NOTE: getDocs is correctly used here to fetch a snapshot based on the query. const querySnapshot = await getDocs(q); if (!querySnapshot.empty) { setUsernameError(`@${sanitizedUsername} is already taken. Choose a different one.`); return false; } setUsernameError(''); return true; } catch (error) { console.error("Error checking username uniqueness:", error); setUsernameError('An unexpected error occurred while checking ID. Please try again.'); return false; } }; // 5. Register the final @user_id const handleRegisterID = async () => { if (!user || !db) return; const uniqueId = usernameInput.toLowerCase().trim(); if (!(await isUsernameUnique(uniqueId))) { return; // Stop if validation failed (error is set by isUsernameUnique) } setLoadingMessage('Registering new ID...'); setAppState('LOADING'); try { // 1. Create a document in the user_id_map for uniqueness check const mapRef = doc(db, USER_ID_MAP_COLLECTION, uniqueId); await setDoc(mapRef, { uid: user.uid, uniqueId: uniqueId, createdAt: new Date().toISOString() }); // 2. Update the user's main profile document const profileRef = doc(db, USER_PROFILES_COLLECTION, user.uid); const profileData = { uniqueId: uniqueId, displayName: user.displayName || 'Anonymous User', email: user.email || 'N/A', pfpUrl: user.photoURL || '', lastLogin: new Date().toISOString() }; await setDoc(profileRef, profileData, { merge: true }); // SUCCESS: Transition to dashboard setProfile({ ...profileData, uniqueId }); setAppState('DASHBOARD'); } catch (error) { console.error("Registration failed:", error); alertUser('Registration Failed', 'Could not save profile. Check the console for details.'); setAppState('CREATE_ID'); // Go back to ID creation on failure } }; // Custom Alert Modal (instead of alert()) const [modal, setModal] = useState({ visible: false, title: '', message: '' }); const alertUser = (title, message) => setModal({ visible: true, title, message }); // --- Rendering Functions --- const renderModal = () => (
